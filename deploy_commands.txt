# Commandes de déploiement Heroku
# Exécuter une par une dans le terminal

# 1. Connexion Heroku
heroku login

# 2. Créer l'application
heroku create iapostemanager

# 3. Configurer les variables d'environnement
heroku config:set FLASK_ENV=production
heroku config:set OPENAI_API_KEY=your_actual_key

# 4. Initialiser Git et déployer
git init
git add .
git commit -m "Deploy IA Poste Manager"
git push heroku main

# 5. Ouvrir l'application
heroku open
